4. Build the CNN model¶

In [9]:
## Defining the Sequential Model
################################################################################
# Define model
model = Sequential()
# Add the first convolutional layer with 16 filters of size (3,3), ReLU
# activation, padding, stride and input shape
model.add(Conv2D(16, kernel_size=(3,3), strides=(2,2), padding='same', \
                 activation='relu', input_shape=(28, 28, 1)))
# Add the first pooling layer to reduce spatial dimensions
model.add(MaxPooling2D(pool_size=(2, 2)))
# Add the second convolutional layer with 16 filters of size (3,3), ReLU
# activation, padding and stride
model.add(Conv2D(16, kernel_size=(3,3), strides=(2,2), padding='same', \
                 activation='relu'))
# Add the second pooling layer to reduce spatial dimensions
model.add(MaxPooling2D(pool_size=(2, 2)))
# Flatten the tensor output from the previous layer
model.add(Flatten())
# Add dense (fully connected) layers
model.add(Dense(32, activation='relu'))       # One Hidden Layer
model.add(Dense(10, activation='softmax'))    # Output Layer
################################################################################
In [10]:
## Compile the model using categorical crossentropy as the loss function and
# Adam as an optimizer
################################################################################
adam = optimizers.Adam(learning_rate=1e-3)
model.compile(loss=losses.categorical_crossentropy, optimizer=adam, \
              metrics=['accuracy'])

                    validation_split=0.1, epochs=20, batch_size=256, verbose=1,\
