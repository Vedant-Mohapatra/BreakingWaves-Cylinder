################################################################################
## Reading the Data
(x_train, y_train), (x_test, y_test) = mnist.load_data()
# Visualizing the first 10 images from the training dataset
print("#######################################################################")
print("First 5 images from the training dataset")
plt.figure(figsize=(10, 5))
for i in range(5):
    plt.subplot(1, 5, i+1)
    plt.imshow(x_train[i], cmap='Greys')
    plt.title(f"Label: {y_train[i]}")
    plt.axis('off')
plt.tight_layout()
plt.show()
# Visualizing the count of different digits in y_train
print(" ")
print("#######################################################################")
print("Count of different digits in y_train")
counts = [0]*10  # initialize a list of 10 zeros, one for each digit
for label in y_train:
    counts[label] += 1
digits = [i for i in range(10)]  # [0, 1, 2, ..., 9]
plt.bar(digits, counts, color='black')
plt.xlabel('Digits')
plt.ylabel('Count')
plt.title('Count of different digits in y_train')
plt.xticks(digits)
plt.grid(axis='y')
plt.show()
################################################################################
## Normalizing the Train and Test Data
# Flatten the images
x_train = x_train.reshape((x_train.shape[0], 28 * 28))
x_test = x_test.reshape((x_test.shape[0], 28 * 28))
# Normalize the data to [0, 1] range
x_train = x_train.astype('float32') / 255.0
x_test = x_test.astype('float32') / 255.0
################################################################################
## Convert to "One-Hot" Vectors Using the to_categorical Function
num_classes = 10
y_train = to_categorical (y_train, num_classes)
y_test_cat=to_categorical (y_test,num_classes)
################################################################################
## Defining the ANN Sequential Model
image_size=28*28
model = Sequential()
model.add(Dense(256, activation='relu', kernel_initializer= 'he_uniform', \
                input_shape=(image_size, ))) #Multiple Dense with Relu activ.
model.add(Dense(8, activation='relu', kernel_initializer='he_uniform'))
model.add(Dense(8, activation='relu' ,kernel_initializer='he_uniform'))
model.add(Dense(num_classes, activation='softmax', kernel_initializer=\
                'he_uniform', input_shape= (image_size,))) # For multi classif.
################################################################################
## Compile model
adam = optimizers.Adam(learning_rate=1e-3)
model.compile(loss=losses.categorical_crossentropy, optimizer=adam, \
              metrics=['accuracy']) # Loss function = Categorical cross entropy
################################################################################
## Summary of Base Model
print(" ")
print("#######################################################################")
print("Summary of the Base Model")
model.summary()
################################################################################
## Fit the Model
print(" ")
print("#######################################################################")
print("Training the ANN Model")
checkpoint = ModelCheckpoint ("model weights.h5", monitor='val accuracy',
                                save_weights_only=True,mode= 'max',verbose=1)
reduce_lr = ReduceLROnPlateau (monitor= 'val_loss', factor=0.1,patience=4,
                                  min_lr=0.00001, model='auto')
callbacks = (checkpoint, reduce_lr)
history=model.fit(x_train, y_train, validation_split=0.1, epochs=20, \
                  batch_size=256, verbose=1, callbacks=callbacks)
################################################################################
## Plot the Accuracy Curve
print(" ")
print("#######################################################################")
print("Accuracy Curve")
loss, accuracy = model.evaluate(x_test,y_test_cat,verbose=False)
plt.plot(history.history['accuracy'])
plt.plot(history.history['val_accuracy' ])
plt.title('model accuracy')
plt.ylabel('accuracy')
plt.xlabel('epoch')
plt.legend(['training','validation'], loc='best')
plt.show()
################################################################################
## Predicting the Model on Test Data
print(" ")
print("#######################################################################")
print("Started Predicting the Model on Test Data")
y_pred=model.predict(x_test)
print("Finished Predicting the Model on Test Data")
# As our outputs are probabilities so we will try to get the output class from
# these probablities by getting the maximum value
y_pred_final=[]
for i in y_pred:
  y_pred_final.append(np.argmax(i))
################################################################################
## Generating Confusion Matrix and Classification Report
cm=confusion_matrix (y_test,y_pred_final)
print(" ")
print("#######################################################################")
print("Confusion Matrix")
plt.figure (figsize= (10,7))
sns.heatmap (cm, annot=True, fmt=' d')
plt.xlabel ('Predicted')
plt.ylabel ('Truth')
plt.show ( )
print(" ")
print("Classification Report")
print (classification_report (y_test,y_pred_final))
